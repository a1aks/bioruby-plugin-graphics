          def arrow(track,direction,x,y,size)
            case direction
            when :right
              track.move_to(x,y)
              track.rel_line_to(size,size)
              track.rel_line_to(-size,size)
              track.close_path.fill
            when :left
              track.move_to(x,y)
              track.rel_line_to(-size,size)
              track.rel_line_to(size,size)
              track.close_path.fill
            when :north
              track.move_to(x-size,y+size)
              track.rel_line_to(size,-size)
              track.rel_line_to(size,size)
              track.close_path.fill
            when :south
              track.move_to(x-size,y-size)
              track.rel_line_to(size,size)
              track.rel_line_to(size,-size)
              track.close_path.fill
            end
          end
